def create
  user = User.find_by_phone_number(sign_in_params[:phone_number])

  if user&.valid_password?(sign_in_params[:password])
    render json: { message: 'Login Successful.' }, status: 200
  else
    render json: {
             message: 'Phone Number or Password is incorrect'
           },
           status: 401
  end
end
